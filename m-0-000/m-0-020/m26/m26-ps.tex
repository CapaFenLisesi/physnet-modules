\Sect{}{}{\SectType{ProblemSet}}{

Problem~8 also occurs in this module's \textit{Model Exam}.

{\bf Note:} Wherever appropriate, take the direction of {\bf stretching} the
spring as the direction of positive displacement.

\begin{one-digit-list}
\item [1.] One end of a horizontal spring is fixed, the other end is found to
be moved 8.0\unit{cm} from its equilibrium position by a force of 20.0\unit{N}.
A mass of 4.0\unit{kg} is then attached to that end of the spring and is pulled
along a frictionless surface to a point 6.0\unit{cm} from the equilibrium position.
The mass is then released at time \m{t = 0} and it executes simple harmonic
motion.
\begin{one-digit-list}
\item [a.] Find the force constant of the spring.
\item [b.] What is the force exerted by the spring on the mass just before it
is released?
\item [c.] Write the equation for the displacement of the mass as a function
of time after it is released.
\item [d.] Calculate the period and frequency of the oscillations.
\end{one-digit-list}

\item [2.] A 3.0\unit{kg} mass hangs from a spring.
A 0.50\unit{kg} body hung below the mass stretches the spring 5.0\unit{cm} farther.
The body is then removed and the mass starts oscillating.
Find the frequency of the motion.

\item [3.] A particle with mass 200.0\unit{g} experiences a force whose associated
potential energy function is \m{E_p = (4.0\unit{J/m\up{2}}) \cdot (x - 0.20\unit{m})^2}.
Find the angular frequency of the motion.

\item [4.] A 50.0\unit{kg} woman standing on the end of a diving board caused it to
be lowered by 30.0\unit{cm} and it then oscillated up and down with her on it.
Assuming the restoring force of the board was linear, calculate the number of
times she went up and down in 2.2\unit{s}.

\item [5.] A car can be considered to be mounted on a vertical spring for
purposes of describing its up and down motion.
If the springs of a \m{1.5 \times 10^3\unit{kg}} car are adjusted so that the
vibrations have a frequency of 3.0\unit{Hz}, what is the spring's force constant?
Find the vibrational frequency if four passengers, averaging 75\unit{kg} each, ride
in the car.

\item [6.] An object hung on a vertical spring stretches it 15\unit{cm}.
The object is then pulled down another 20.0\unit{cm} and released.
\begin{one-digit-list}
\item [a.] Find the displacement equation for the object.
\item [b.] Find the velocity and acceleration as functions of time.
\end{one-digit-list}

\item [7.] A vertically hanging spring has a force constant of 150.0\unit{N/m}.
A 5.0\,kg mass is now hung from the end of the spring and the mass settles
down to a new equilibrium position \m{x_0}.
Finally, the mass is displaced 0.20\unit{m} below its equilibrium position and
given an initial velocity of 2.0\unit{m/s} downward.
\begin{one-digit-list}
\item [a.] Write Hooke's law for the mass-spring system and calculate the
displacement when the force is 30.0\unit{N}.
\item [b.] Find the displacement equation.
\item [c.] Calculate the period.
\end{one-digit-list}

\item [8.] An object undergoing SHM has a mass of 2.0\unit{kg} and force
constant 60.0\unit{N/m}.
Its equilibrium position is at \m{x_0 = 0.30\unit{m}}.
\begin{one-digit-list}
\item [a.] Write Hooke's law and find the position of the object when the
force on it is 25\unit{N}.
\item [b.] Find the displacement equation if the object: (1) has zero
velocity when it is at the point \m{x = 0.80\unit{m}}; and (2) at time zero is at
the equilibrium position and is heading toward negative displacements.
Then find \m{x} at \m{t = 1.0\unit{s}}.
\item [c.] Calculate the period of the oscillations.
\end{one-digit-list}
\end{one-digit-list}

\BriefAns

\begin{one-digit-list}
\item [1.] a. \m{k = \dfrac{-F}{x - x_0} =
-\dfrac{(-20\,{\unit{N}})}{0.08\unit{m}} = 250\unit{N/m}}
\begin{one-digit-list}
\item [b.] \m{F = - k (x - x_0) = - (250\unit{N/m})(0.06\unit{m}) = - 15\unit{N}}
\item [c.] \m{\omega =
           \left( \dfrac{250\unit{N/m}}{4.0\unit{kg}}\right)^{1/2} = 7.9\unit{rad/s}}
\item []   \m{x(t) = 0.060}\,m \m{\cos\left( \dfrac{7.9}{\unit{s}} t \right)}
\item [d.] \m{T = \dfrac{2 \pi}{\omega} = \dfrac{2 \pi}{7.9/{\unit{s}}} = 0.80\unit{s}};
           \m{\nu = \dfrac{1}{T} = 1.3\unit{Hz}}
\end{one-digit-list}

\item [2.] \m{k = \dfrac{-F}{x - x_0} = \dfrac{m g}{x - x_0} =
              \dfrac{(0.50\unit{kg})(9.8\unit{m/s\up{2}})}{0.05\unit{m}} = 98\unit{N/m}}
\item []   \m{\nu = \dfrac{1}{2 \pi} \left( \dfrac{98\unit{N/m}}{3.0\unit{kg}}\right)^{1/2} = 0.91\unit{Hz}}

\item [3.] \m{E_p = \dfrac{k}{2} \left( x-x_0 \right)^2 = 4.0\unit{J/m\up{2}} \left( x - x_0 \right)^2}
\item []   \m{\dfrac{k}{2} = 4.0\unit{J/m\up{2}} \Rightarrow k = 8.0\unit{N/m}}
\item []   \m{\omega = \left( \dfrac{8.0\unit{N/m}}{0.20\unit{kg}} \right)^{1/2} = 6.3\unit{rad/s}}

\item [4.] \m{k = \dfrac{-F}{x - x_0} =
\dfrac{mg}{x - x_0} = \dfrac{(50\unit{kg})(9.8\unit{m/s\up{2}})}{0.30\unit{m}} = 1600\unit{N/m}}
\item []   \m{T = 2 \pi \left( \dfrac{50\unit{kg}}{1600\unit{N/m}}\right)^{1/2} = 1.1\unit{s}}

\item []   \m{\# = \dfrac{2.2\unit{s}}{T} = \dfrac{2.2\unit{s}}{1.1\unit{s}} = 2 \text{ times}}

\item [5.] \m{k=m(2\pi\nu)^2=(1500\unit{kg})\left[2\pi(3.0\unit{Hz})\right]^2 = 5.3 \times 10^5\unit{N/m}}
\item []   \m{\nu=\dfrac{1}{2\pi}\left(\dfrac{k}{m_\text{total}}\right)^{1/2}=\dfrac{1}{2\pi}
              \left(\dfrac{5.3 \times 10^5\unit{N/m}}{1800\unit{kg}} \right)^{1/2} = 2.7\unit{Hz}}

\item [6.] \m{x_0 = \dfrac{mg}{k} \Rightarrow \dfrac{k}{m} = \dfrac{g}{x_0} =
              \dfrac{9.8\unit{m/s\up{2}}}{0.15\unit{m}}}
\item []   \m{\omega=\left(\dfrac{9.8\unit{m/s\up{2}}}{0.15\unit{m}} \right)^{1/2} = 8.1\unit{rad/s}}
\begin{one-digit-list}
\item [a.] \m{x - 0.15\unit{m} = 0.20\unit{m} \cos\left( \dfrac{8.1}{\unit{s}} t \right)}
\item [b.] \m{v=\dfrac{dx}{dt}=(0.20\unit{m})(8.1/\unit{s})
                                   \left[-\sin\left(\dfrac{8.1}{\unit{s}}t\right)\right]}
\item []   \m{a = \dfrac{dv}{dt} = (-1.6\unit{m/s})
              \left( \dfrac{8.1}{\unit{s}} \right)\cos\left( \dfrac{8.1}{\unit{s}} t \right) =
    - 13\unit{m/s\up{2}} \cos\left( \dfrac{8.1}{\unit{s}} t \right).}
\end{one-digit-list}

\item [7.] \m{x_0=\dfrac{mg}{k}=\dfrac{(5.0\unit{kg})(9.8\unit{m/s\up{2}})}{150\unit{N/m}}=0.33\unit{m}}
\begin{one-digit-list}
\item [a.] \m{F = (-150\unit{N/m})(x - 0.33\unit{m})}
\item []   displacement \m{ = (x - 0.33\unit{m}) = \dfrac{-30\unit{N}}{-150\unit{N/m}} = 0.20\unit{m}}
\item [b.] \m{\omega = \left( \dfrac{150\unit{N/m}}{5.0\unit{kg}} \right)^{1/2} = 5.5\unit{rad/s}}
\item []   \m{x - 0.33\unit{m} = A \cos\left( \dfrac{5.5}{\unit{s}} t + \delta_0 \right)}
\item []   \m{v=\dfrac{dx}{dt}=-A(\dfrac{5.5}{\unit{s}})\sin\left(\dfrac{5.5}{\unit{s}}t+\delta_0 \right)}
\item []   at \m{t = 0}: \m{0.20\unit{m} = A \cos\delta_0}
\item []   \m{2\unit{m/s} = - A \left( \dfrac{5.5}{\unit{s}} \right) (\sin \delta_0)}
\item []   \m{(0.20\unit{m})^2 + \left( \dfrac{2\unit{m/s}}{5.5/\unit{s}} \right)^2 =
              A^2 (\cos^2\delta_0 + \sin^2\delta_0) = A^2}
\item []   \m{A = 0.42\unit{m}}
\item []   \m{\delta_0=\cos^{-1}\left(\dfrac{0.20\unit{m}}{0.42\unit{m}}\right)=62\degrees=1.1\unit{rad}}
\item []   \m{x-0.33\unit{m}=0.42\unit{m} \cos\left( \dfrac{5.5}{\unit{s}} t + 1.1\,{\unit{rad}} \right)}
\item [c.] \m{T = \dfrac{2 \pi}{\omega} = \dfrac{2 \pi}{5.5/{\unit{s}}} = 1.1\unit{s}}
\end{one-digit-list}

\item [8.] a. \m{F = (- 60\unit{N/m}) (x - 0.30\unit{m})}
\begin{one-digit-list}
\item []   \m{F = 25\unit{N}}; \m{x = - 0.12\unit{m}}
\item [b.] \m{x-0.30\unit{m} = 0.50\unit{m} \cos\left(\dfrac{5.5}{\unit{s}} t + \dfrac{\pi}{2}\right) };
           \m{x(1.0\unit{s}) = 0.65\unit{m}}
\item [c.] \m{T = 1.1\unit{s}}
\end{one-digit-list}
\end{one-digit-list}
