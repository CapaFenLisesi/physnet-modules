\begin{one-digit-list}
\item [1.] A harmonic oscillator of mass 0.64\,kg has a
displacement from equilibrium given by:
%
\Eqn{}{x = (2.1\unit{m})\,\cos{[(1.25/\unit{s})t + (\pi/4)]}\,.}
%
\begin{one-digit-list}
\item [a.] Draw a phasor diagram showing the displacement, velocity, and
acceleration phasors of the oscillator, and the phasor representing the
restoring force acting on it, at time \m{t = 0}.
\item [b.] Draw another phasor diagram of the four phasors at time
\m{t = 1.0\unit{s}}.
Describe how the phasor diagram changes during the time interval between
\m{t =0} and \m{t = 1.0\unit{s}}.
\end{one-digit-list}

\item [2.] Draw a phasor diagram showing the displacement, velocity,
acceleration and restoring force phasors at \m{t = P/2} where \m{P} is period for
the case:
%
\Eqn{}{x(t) = A\,\cos{[(\pi/s)t + \pi/2)]}\,.}
%
Also, show the total phase and mark its value.
\end{one-digit-list}
