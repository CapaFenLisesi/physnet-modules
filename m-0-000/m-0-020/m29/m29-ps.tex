\Sect{}{}{\SectType{ProblemSet}}{

\begin{one-digit-list}
\item [1.] A meter needle is to be designed so that its amplitude of vibration
will drop by 99.0\% in three oscillations, hence be unobservable after that.
If the frequency of oscillation of the needle is 2.00 per second, determine
the design damping constant \m{\gamma}.

\item [2.] An engineering student observing the diving board at the swimming pool
notices that the vibrating board's amplitude changes from 14.0\,cm to 8.00\,cm
in 0.800 seconds.
Determine the damping constant \m{\gamma} of the board.
\end{one-digit-list}

\BriefAns
\begin{one-digit-list}
\item [1.] \m{\gamma} = 3.07\unit{/s}

\item [2.] \m{\gamma} = 0.700\unit{/s}
\end{one-digit-list}
